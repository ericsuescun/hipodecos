      CASE M.PATOLOGO = "MAS"
        USE Cbio
go top
        BROWSE WIDTH 12 FIELDS NUMERO, nombre, nombre2, apellido, apellido2, CEDULA, edad, oficina, bloque, imprimir, DESCR, DIAGNOSTIC, FECHA, patologo, secretaria, DX FOR patologo = "MAS" .AND. empty(imprimir) .and. not empty(fecha)
        go top
        REPLACE REVISOR WITH M.REVISOR FOR patologo = "MAS" .and. not empty(fecha) .and. imprimir = 1 .or. patologo = "MAS" .and. not empty(fecha) .and.imprimir = 2
        go top
        REPORT FORM FOTOS TO PRINT NOCONSOLE FOR NOT EMPTY(FOTO) .AND. imprimir = 1
        REPORT FORM IMPRI TO PRINT NOCONSOLE FOR EMPTY(FOTO) .AND. imprimir = 1
        go top
        REPLACE imprimir with 2 for imprimir = 1
        close data
