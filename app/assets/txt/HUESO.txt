OPC = 0
DEFINE WINDOW MARCO5 FROM 03,06 TO 09,70 DOUBLE
ACTIVATE WINDOW MARCO5
@ 02,02 say "ANOTE EL CODIGO CORRESPONDIENTE: .."GET OPC
READ
IF OPC = 0
   _CUROBJ = OBJNUM(DESCR)
   DEACTIVATE WINDOW MARCO5
RELEASE WINDOW MARCO5

   RETURN
   ELSE
DO CASE
   CASE OPC = 5491000
        APPEND MEMO DESCR FROM 54i1
   APPEND MEMO DIAGNOSTIC FROM 54i1dx
      REPLACE CODIGO5 WITH "2139"
   DO CASE
      CASE EMPTY(CODIGO)
      REPLACE CODIGO  WITH "54i1"
      CASE .NOT. EMPTY(CODIGO) .AND. EMPTY(CODIGO1)
      REPLACE CODIGO1 WITH "54i1"
      CASE .NOT. EMPTY(CODIGO1) .AND. EMPTY(CODIGO2)
      REPLACE CODIGO2 WITH "54i11"

      REPLACE CODIGO3 WITH "54i1"
      CASE .NOT. EMPTY(CODIGO3) .AND. EMPTY(CODIGO4)
      REPLACE CODIGO4 WITH "54i1"
   ENDCASE   
   
   CASE OPC = 5492000
        APPEND MEMO DESCR FROM 54i2
   APPEND MEMO DIAGNOSTIC FROM 54i2dx
      REPLACE CODIGO5 WITH "2139"
   DO CASE
      CASE EMPTY(CODIGO)
      REPLACE CODIGO  WITH "54i2"
      CASE .NOT. EMPTY(CODIGO) .AND. EMPTY(CODIGO1)
      REPLACE CODIGO1 WITH "54i2"

      REPLACE CODIGO2 WITH "54i2"

      REPLACE CODIGO3 WITH "54i2"
      CASE .NOT. EMPTY(CODIGO3) .AND. EMPTY(CODIGO4)
      REPLACE CODIGO4 WITH "54i2"
   ENDCASE   
   
   CASE OPC = 5495000
        APPEND MEMO DESCR FROM 54i5
   APPEND MEMO DIAGNOSTIC FROM 54i5dx

      CASE EMPTY(CODIGO)
      REPLACE CODIGO  WITH "54i5"

      REPLACE CODIGO1 WITH "54i5"

      REPLACE CODIGO3 WITH "54i5"

      REPLACE CODIGO4 WITH "54i5"
   ENDCASE   
ENDCASE
ENDIF
